def isCycle(v,adj):
    queue = [[v,-1]]  
    while len(queue) > 0:
        node,parent = queue.pop(0)
        visited[node] = True
        for neighbour in adj[node]:
            if not visited[neighbour]:
                queue.append([neighbour,node])
            elif neighbour != parent:
                return True
    return False

ans = False
for i in range(V):

        continue
